import csv
import random

with open("fitness_data.csv","w", newline="") as file:
    writer=csv.writer(file)
    writer.writerow(["day", "steps", "calories", "sleep_hours"])

    for day in range(1,31):
        steps= random.randint(1000,20000)
        calories= random.randint(100,1000)
        sleep= round(random.uniform(4.0,10.0),1)

        writer.writerow([day,steps,calories,sleep])

print("CSV file with random data created.")
